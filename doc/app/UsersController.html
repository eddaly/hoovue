def update
  @user = User.find(params[:id])
  respond_to do |format|
    if @user.update_attributes(params[:user])
      format.html { redirect_to @user, notice: 'Your Profile has been updated.' }
    else
      format.html { render action: "edit" }
    end
  end
end
